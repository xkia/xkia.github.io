批量更新 ​

python
import psycopg
from psycopg.extras import execute_batch

update_sql = f"""
        UPDATE table_name 
        SET column1  = %s, column2 = %s 
        WHERE id = %s
    """

    # 数据处理：将每个更新的项包装成一个元组 (column1, column2, id)
    update_data = [
        (item["column1"], item["column2"], item["id"])
        for item in data
    ]

    try:
        with conn.cursor() as cur:
            # 使用 execute_batch 批量执行更新
            execute_batch(
                cur, update_sql, update_data, page_size=1000
            )  # 可以调整 page_size 大小来控制每批次更新的记录数

            # 提交事务
            conn.commit()
            logging.info(
                f"更新成功行数: {len(data)}"
            )

    except Exception as e:
        conn.rollback()
        logging.error(f"更新失败: {e}")

查询并获取列名 ​

返回的数据接口为dict结构，可直接使用字典的方式获取对应的value值

python
import psycopg
from psycopg.extras import DictCursor

with conn.cursor(cursor_factory=DictCursor) as cur:
    cur.execute("SELECT id, name, age FROM table_name")
    results = cur.fetchall()

    if results:
        column_names = results[0].keys()
        print("Column Names:", column_names)

    for row in results:
        print(dict(row))

# 返回示例
{'id': 1863530930946191361, 'name': 'name'}
